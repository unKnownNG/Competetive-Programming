n = int(input())
for _ in range(n):
    x = int(input())
    s = input()
    l = s[0]
    r = s[-1]
    f = 0
    d = {}
    for i in range(1, x - 1):
        if s[i] not in d:
            d[s[i]] = 1
        else:
            f  = 1
            continue
    if l in d or r in d or f:
        print("yes")
    else:print("no")
